main:
    main:dec-def*

dec-def:
    func-def:type IDEN LPAREN params-def? RPAREN LBRACE line* ret=(RETURN expr SEMI)? RBRACE
    dec-assign:dec-assign

line:
    control:control
    statement:statement? (COMMA statement)*

control:
    if:IF LPAREN expr RPAREN LBRACE line* RBRACE

params-def:
    params:param-def? (COMMA param-def)*
    
param-def:
    param:type variable

value:
    int:NUM
    variable:variable
    str:STR
    char:CHAR
    cast:LPAREN type RPAREN value
    comp-lit:LBRACE expr? (COMMA expr)* RBRACE

variable:
    variable:IDEN (LSQUARE NUM? RSQUARE)*

expr:
    sub-expr:sub-expr
    comp:comp-expr

sub-expr:
    value:value
    parens:LPAREN expr RPAREN
    func-call:IDEN LPAREN expr? (COMMA expr)* RPAREN

comp-expr:
    op:sub-expr (PLUS | MINUS | DIVIDE | MULTIPLY | EEQUALS) expr
    is-equal:sub-expr EEQUALS expr

statement:
    dec-assign:dec-assign
    var-dec:type variable
    assign:variable EQUALS expr
    expr:expr

dec-assign:
    standard:type variable EQUALS expr

type:
    int:INT_TYPE
    char:CHAR_TYPE
    custom:IDEN
    pointer:ASTERISKS type
