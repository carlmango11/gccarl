main:
  main:include* dec-def*

include:
    include:"#include" include-iden

include-iden:
    std-iden:"<" IDEN ">"
    custom-iden:"\"" IDEN "\""

dec-def:
    func-def:type IDEN "(" params-def? ")" "{" statement-comma* return? "}"
    dec-assign:dec-assign

return:
    return:"return" expr ";"

statement-comma:
    statement:statement ";"

params-def:
    params:param-def comma-param-def*

comma-param-def:
    param:"," param-def
    
param-def:
    param:type IDEN

value:
    int:NUM
    variable:variable

variable:
    variable:IDEN array-index?

array-index:
    array-index:"[" NUM "]"

expr:
    add:value "+" expr
    sub:value "-" expr
    parens:"(" expr ")"
    value:value

statement:
    dec-assign:dec-assign
    assign:variable "=" expr
    func-call:IDEN "(" params? ")"

dec-assign:
    assign:type variable "=" expr

params:
    params:expr comma-expr*

comma-expr:
    comma-expr:"," expr

type:
    int:"int"
    char:"char"
    long:"long"
