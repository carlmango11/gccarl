main:
  main:dec-def*

dec-def:
    func-def:type IDEN LPAREN params-def? RPAREN LBRACE statement-comma* return? RBRACE
    dec-assign:dec-assign

return:
    return:RETURN expr SEMI

statement-comma:
    statement:statement SEMI

params-def:
    params:param-def comma-param-def*

comma-param-def:
    param:COMMA param-def
    
param-def:
    param:type variable

value:
    int:NUM
    variable:variable
    str:STR
    char:CHAR
    cast:LPAREN type RPAREN value

comma-expr:
    comma-expr:COMMA expr

variable:
    variable:IDEN array-index*

array-index:
    array-index:LSQUARE NUM? RSQUARE

expr:
    sub-expr:sub-expr
    comp:comp-expr

sub-expr:
    value:value
    parens:LPAREN expr RPAREN
    func-call:IDEN LPAREN params? RPAREN

comp-expr:
    add:sub-expr "+" expr
    sub:sub-expr "-" expr

statement:
    dec-assign:dec-assign
    var-dec:type variable
    assign:variable EQUALS expr
    expr:expr

dec-assign:
    standard:type variable EQUALS expr
    array:type variable EQUALS composite-literal

composite-literal:
    array-val:LBRACE comp-entries? RBRACE

comp-entries:
    entries:expr comma-expr*

params:
    params:expr comma-expr*

comma-expr:
    comma-expr:COMMA expr

type:
    int:INT_TYPE
    char:CHAR_TYPE
    custom:IDEN
    pointer:ASTERISKS type
