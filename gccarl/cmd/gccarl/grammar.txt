main:
  main:include* dec-def*

include:
    include:HASH INCLUDE include-iden

include-iden:
    std-iden:LESS_THAN IDEN GREATER_THAN
    custom-iden:DQUOTE IDEN DQUOTE

dec-def:
    func-def:type IDEN LPAREN params-def? RPAREN LBRACE statement-comma* return? RBRACE
    dec-assign:dec-assign

return:
    return:RETURN expr SEMI

statement-comma:
    statement:statement SEMI

params-def:
    params:param-def comma-param-def*

comma-param-def:
    param:COMMA param-def
    
param-def:
    param:type variable

value:
    int:NUM
    variable:variable
    str:STR
    char:QUOTE CHAR QUOTE
    array:array-val
    cast:LPAREN type RPAREN value

array-val:
    array-val:LBRACE array-entries? RBRACE

array-entries:
    entries:expr comma-expr*

comma-expr:
    comma-expr:COMMA expr

variable:
    variable:IDEN array-index?

array-index:
    array-index:LSQUARE NUM? RSQUARE

expr:
    sub-expr:sub-expr
    comp:comp-expr

sub-expr:
    value:value
    parens:LPAREN expr RPAREN
    func-call:IDEN LPAREN params? RPAREN

comp-expr:
    add:sub-expr "+" expr
    sub:sub-expr "-" expr

statement:
    dec-assign:dec-assign
    var-dec:type variable
    assign:variable EQUALS expr
    expr:expr

dec-assign:
    assign:type variable EQUALS expr

params:
    params:expr comma-expr*

comma-expr:
    comma-expr:COMMA expr

type:
    int:INT
    char:CHAR
    custom:IDEN
    pointer:ASTERISKS type
