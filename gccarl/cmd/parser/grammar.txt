main:
  main:dec-def*

dec-def:
    func-def:type IDEN LPAREN params-def? RPAREN LBRACE line* RBRACE
    dec-assign:dec-assign

line:
    control:control
    statement:statement-comma

control:
    if:IF LPAREN expr RPAREN LBRACE block-or-line RBRACE else?
    while:WHILE LPAREN expr RPAREN LBRACE line* RBRACE

block-or-line:
    block:LBRACE line* RBRACE
    line:line

else:
    else:ELSE block-or-line

statement-comma:
    statement:statement SEMI

params-def:
    params:param-def comma-param-def*

comma-param-def:
    param:COMMA param-def

param-def:
    param:type variable

value:
    int:NUM
    variable:variable
    str:STR
    char:CHAR
    cast:LPAREN type RPAREN value
    comp-lit:composite-literal

comma-expr:
    comma-expr:COMMA expr

variable:
    variable:IDEN array-index*

array-index:
    array-index:LSQUARE NUM? RSQUARE

expr:
    sub-expr:sub-expr
    comp:comp-expr

sub-expr:
    value:value
    parens:LPAREN expr RPAREN
    func-call:IDEN LPAREN params? RPAREN

comp-expr:
    comp-expr:sub-expr operator expr

operator:
    less:LESS_THAN
    equal:EEQUALS
    plus:PLUS
    minus:MINUS

statement:
    dec-assign:dec-assign
    var-dec:type variable
    assign:variable EQUALS expr
    expr:expr
    return:RETURN expr

dec-assign:
    standard:type variable EQUALS expr

composite-literal:
    array-val:LBRACE comp-entries? RBRACE

comp-entries:
    entries:expr comma-expr*

params:
    params:expr comma-expr*

comma-expr:
    comma-expr:COMMA expr

type:
    int:INT_TYPE
    char:CHAR_TYPE
    void:VOID
    custom:IDEN
    pointer:ASTERISKS type
