main:
  main:include* dec-def*

include:
    include:"#include" include-iden

include-iden:
    std-iden:"<" IDEN ">"
    custom-iden:"\"" IDEN "\""

dec-def:
    func-def:type IDEN "(" params-def ")" "{" statement-comma* "}"
    dec-assign:dec-assign

statement-comma:
    statement:statement ";"

params-def:
    none:
    params:param-def comma-param-def*
    
param-def:
    param:type IDEN

value:
    int:NUM
    variable:IDEN

expr:
    add:value "+" expr
    sub:value "-" expr
    parens:"(" expr ")"
    value:value

statement:
    dec-assign:dec-assign
    assign:IDEN "=" expr
    func-call:IDEN "(" params ")"

dec-assign:
    assign:type IDEN "=" NUM

params:
    none:
    params:IDEN comma-iden*

comma-iden:
    comma-iden:"," IDEN

type:
    int:"int"
    char:"char"
    long:"long"
